comandos

# Instala Angular CLI si no lo tienes:
npm install -g @angular/cli

# Crea el proyecto y entra en la carpeta
ng new proyecto-etxebizitzak --routing=true --style=css
cd proyecto-etxebizitzak

# Instala JSON-Server
npm install json-server --save-dev

# Crea una carpeta para servicios
mkdir src/app/services

# Servicio para Propiedades 
ng generate service services/property

# Servicio para Contacto
ng generate service services/contact

# Servicio para Imágenes 
ng generate service services/property-images

# Crea una carpeta para las páginas/componentes
mkdir src/app/components

# Componentes de páginas
ng generate component components/home-page
ng generate component components/property-page
ng generate component components/property-edit-page
ng generate component components/property-create-page
ng generate component components/contact-page
ng generate component components/filtered-list-page

# Componentes reutilizables (ej. la cabecera)
ng generate component components/header

# --- EJECUCIÓN (En terminales separadas) ---

# (Terminal 1) Ejecuta el frontend de Angular
npx ng serve --open

# (Terminal 2) Ejecuta el mock backend (asegúrate de tener db.json)
npx json-server --watch db.json

# bootstrap
npm install bootstrap @popperjs/core bootstrap-icons

# mapbox
npm install mapbox-gl
npm install --save-dev @types/mapbox-gl

# ngx-translate
npm install @ngx-translate/core @ngx-translate/http-loader
